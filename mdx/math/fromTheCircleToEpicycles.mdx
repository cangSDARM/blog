function Sinusoid(s, vC, vR, angle, ampl, freq, phase, color, tf, tr) {
    this.s = s;
    this.vC = vC;
    this.vR = vR;
    this.ampl = ampl;
    this.freq = freq;
    this.phase = phase;
    this.color = color;
    this.angle = angle;
    this.tf = tf;
    this.tr = tr;
}

const SinusoidPrototype = {
    draw() {
        // Actual Circle
        this.s.push();
        this.s.noFill();
        this.s.stroke(this.color);
        this.s.circle(this.vC.x, this.vC.y, 2*this.ampl*this.tr);
        this.s.pop();
        // Moving Radius
        this.s.push();
        this.s.stroke(theme.radiusColorLight);
        this.s.line(this.vC.x, this.vC.y, this.vR.x, this.vR.y);
        this.s.pop();
    },
    update(fromEpi) {
        if (fromEpi!==undefined) {
            this.vC.x = fromEpi.vR.x;
            this.vC.y = fromEpi.vR.y;
        }
        this.vR.x = this.vC.x + this.ampl * this.s.sin(this.angle * this.freq + this.phase + this.s.HALF_PI) * (this.tr);
        this.vR.y = this.vC.y + this.ampl * this.s.cos(this.angle * this.freq + this.phase + this.s.HALF_PI) * (this.tr);
        this.angle+=this.tf;
    }
}

// Complex numbers
function Complex(re, im) {
    this.re = re;
    this.im = im;
}
const ComplexPrototype = {
    add(cn) {
        this.re += cn.re;
        this.im += cn.im;
    },
    prd(cn) {
        return new Complex(
            this.re * cn.re - this.im * cn.im,
            this.re * cn.im + this.im * cn.re
        );
    }
}
Object.assign(Complex.prototype, ComplexPrototype);

// DFT
const addDft = (s) => {
    s.dft = (vals) => {
        let coefs = [];
        for(let i = 0; i < vals.length; ++i) {
            let sum = new Complex(0, 0);
            for(let j = 0; j < vals.length; ++j) {
                let phi = 2 * s.PI * i * j / vals.length;
                sum.add(vals[j].prd(new Complex(s.cos(phi), -s.sin(phi))));
            }
            sum.re = sum.re / vals.length;
            sum.im = sum.im / vals.length;
            coefs[i] = {
                freq: i,
                amp: s.sqrt(sum.re * sum.re + sum.im * sum.im),
                phase: s.atan2(sum.im, sum.re)
            };
        }
        return coefs;
    },
    s.dftFreqSort = (vals) => {
        return s.dft(vals).sort((a, b) => b.freq - a.freq);
    }
    s.dftAmpSort = (vals) => {
        return s.dft(vals).sort((a, b) => b.amp - a.amp);
    }
}

const addEpiCycles = (s) => {

### 简单圆

圆以圆心$P(a,b)$和半径$r$表示，代数表示为：$(x-a)^2+(y-b)^2=r^2$

圆心在原点($a=0, b=0$)时，称为单位圆。代数表示为：$x^2+y^2=1$

const simpleCircleRotating = (s) => {
    addPaintGrid(s);
    addShowFps(s);

    const d = 200;
    const r = d / 2;
    const f = theme.frequency;
    let rColor;

    let angl = s.HALF_PI;
    const reset = s.TWO_PI + s.HALF_PI;

    let vC, vR, vTxtAb, cBuff;

    s.setup = () => {
        const canvas = s.createCanvas(theme.canvasX, theme.canvasY);
        canvas.parent("simple-circle-rotating-sketch");

        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                entry.isIntersecting ? s.loop() : s.noLoop();
            });
        }, { threshold: 0.1 });
        observer.observe(canvas.elt);

        s.textFont(theme.textFont);
        s.frameRate(theme.frameRate);
        rColor = s.color(theme.radiusColorLight);

        vC = s.createVector(s.width / 2, s.height / 2);
        vR = s.createVector(0, 0);
        vTxtAb = s.createVector(vC.x - 12, vC.y + 12);

        cBuff = s.createGraphics(s.width, s.height);
        cBuff.noFill();
        s.paintGrid(cBuff, s.width, s.height, vC, r / 5, 5, {
            showUnits: true, showOrigin: true, showY: true, showX: true
        });
        cBuff.circle(vC.x, vC.y, d);
    };

    s.draw = () => {
        s.background(theme.bkgColor);
        s.image(cBuff, 0, 0);

        const sSin = s.sin(angl);
        const sCos = s.cos(angl);

        vR.set(vC.x + sSin * r, vC.y + sCos * r);

        s.stroke(rColor);
        s.line(vC.x, vC.y, vR.x, vR.y);
        s.circle(vR.x, vR.y, 3);

        s.noStroke();
        // CHANGED: Using theme colors instead of hardcoded 255
        s.fill(theme.radiusColor || 0);

        const unitX = sSin;
        const unitY = -sCos;

        s.text(`(x=${unitX.toFixed(2)}, y=${unitY.toFixed(2)})`, vR.x + 5, vR.y);
        s.text("P(a,b)", vTxtAb.x, vTxtAb.y - 20);
        s.text("r (radius)", (vC.x + vR.x) / 2, (vC.y + vR.y) / 2);

        // Identity sum at bottom
        const sqrX = (unitX * unitX).toFixed(2);
        const sqrY = (unitY * unitY).toFixed(2);
        s.text(`x² + y² = ${sqrX} + ${sqrY} = 1.00`, 10, s.height - 10);

        angl += f;
        if (angl > reset) angl = s.HALF_PI;

        s.showFps();
    };
};

#### $\pi$

通常使用弧度$\pi$表示角度，而不是度数$°$。圆周：$P=2\pi r$，面积：$A=\pi r^2$

$\pi$是无理数(Irrational)以及超越数(Transcendental)

const rotatingPi = (s) => {

    addShowFps(s);
    addPaintGrid(s);
    addPauseLoop(s);

    const w = 425;
    const h = 175;
    const d = 100;
    const r = d / 2;
    const f = theme.frequency;
    const xOff = r;
    const twoPiR = s.TWO_PI * r;

    let ang = 0;
    let vC, vM;
    let buff;
    let isResetting = false;

    s.setup = () => {
        const canvas = s.createCanvas(w, h);
        canvas.parent('rotating-PI-sketch');
        s.textFont(theme.textFont);
        s.frameRate(theme.frameRate);

        // Initialise vectors
        vC = s.createVector(xOff, h / 2);
        vM = s.createVector(vC.x + s.sin(ang) * r, vC.y + s.cos(ang) * r);

        // Circle Buffer
        buff = s.createGraphics(w, h);
        let vPgo = s.createVector(vC.x, vC.y + r);
        let pGProps = {
            showUnits: true,
            showOrigin: true,
            showY: false,
            showX: true
        };
        s.paintGrid(buff, s.width, s.height, vPgo, r / 2, 2, pGProps);

        // --- Intersection Observer ---
        // We place this at the end of setup so canvas.elt is definitely ready
        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                // Only start looping if the sketch is on screen
                // AND we aren't currently in the middle of a 3-second reset pause
                if (entry.isIntersecting && !isResetting) {
                    s.loop();
                } else {
                    s.noLoop();
                }
            });
        }, { threshold: 0.05 }); // Triggers when 5% is visible

        observer.observe(canvas.elt);
    };

    s.draw = () => {
        s.background(theme.bkgColor);
        s.image(buff, 0, 0);

        // 1. Static-style elements (Rolling circle outline)
        s.noFill();
        s.stroke(theme.lightCircleColor || 200);
        s.circle(vC.x, vC.y, d);

        // 2. Path traveled line (Green line on x-axis)
        s.stroke(theme.thetaColor);
        s.strokeWeight(3);
        s.line(xOff, h / 2 + r, vC.x, h / 2 + r);

        // 3. Moving point
        s.fill(theme.thetaColor);
        s.noStroke();
        s.circle(vM.x, vM.y, 5);

        // Update logic (only if not in reset phase)
        if (!isResetting) {
            vC.x += f * r;
            ang -= f;
            vM.set(vC.x + s.sin(ang) * r, vC.y + s.cos(ang) * r);
        }

        // Reset Logic
        if (vC.x > twoPiR + xOff) {
            isResetting = true;

            // Display 2π label
            s.fill(theme.thetaColor);
            s.text("2π", xOff + s.PI * r, h / 2 + r - 15);

            s.noLoop(); // Pause the animation clock

            setTimeout(() => {
                vC.x = xOff;
                ang = 0;
                vM.set(vC.x + s.sin(ang) * r, vC.y + s.cos(ang) * r);
                isResetting = false;

                // Re-start the loop ONLY if it's still visible
                // (The observer will have handled the state if we scrolled away)
                s.loop();
            }, 3000);
        }

        s.showFps();
    };
}

#### 奇偶性

在数学中，奇偶性(Parity)描述的是函数的对称性。根据函数在其输入值的符号反转时的行为，可将其分为偶函数、奇函数，或二者皆非。

- 余弦函数是偶函数：$\cos(x)=\cos(-x)$
- 正弦函数是奇函数：$\sin(-x)=-\sin(x)$，或$\sin(x)=-\sin(-x)$

### 复数平面

将一个复数乘以虚数单位$i$，对应于在复平面上将其逆时针旋转：$\frac{\pi}{2}(90^\circ)$。

这个旋转发生在一个以原点为圆心的圆上，该圆的半径就是该复数的模(复数的长度)：$r=\lvert a+bi\rvert=\sqrt{a^2+b^2}$

从几何角度看，乘以$i$保持模不变，只使辐角(复数在复平面上与实轴正方向的夹角)增加$\frac{\pi}{2}$

### $e$

自然指数函数通常写作：$f(x) = e^x$，它是指数函数的一种特殊形式，其底数为常数$e$，即欧拉数

$e$也是无理数以及超越数

$e$与以下等式等价

$$
\begin{gather*}
e=\sum_{n=0}^\infty \frac{1}{n!}=\frac{1}{n!}+\frac{1}{1!}+\frac{2}{2!}+\dots \\
e=\lim_{x \to \infty} (1+\frac{1}{x})^x \\
e=\lim_{x \to 0} (1+x)^{\frac{1}{x}}
\end{gather*}
$$

对$e$求导，满足：$\frac{d}{dx} e^{ax} = ae^{ax}$

在复平面上考虑$a=i$，则有：$f(x)=e^{ix}$，及

$$
\begin{align*}
\frac{d}{dx} f(x) &= ie^{ix} \\
\frac{d^2}{dx^2} f(x) &= -e^{ix} \\
\frac{d^3}{dx^3} f(x) &= -ie^{ix} \\
\frac{d^4}{dx^4} f(x) &= e^{ix} = f(x)
\end{align*}
$$

对$e^{ix}$每求导一次，就相当于给该函数乘以一个$i$，因此求导运算本身就起到了旋转算子的作用

### 欧拉恒等式

具有恒定模数且辐角线性增长的复值函数，必然具有形式：$e^{ix}=\cos(x)+i\sin(x)$。
即欧拉恒等式

选定$x=\pi$，则有：$e^{i\pi}=-1+0i \to e^{i\pi}+1=0$

取$x=-x$，并加上或减去$x=x$的原式，则可得出

$$
\begin{align*}
\cos(x) &= \frac{e^{ix}+e^{-ix}}{2} \\
\sin(x) &= \frac{e^{ix}-e^{-ix}}{2i} \\
\end{align*}
$$

则在复平面上单位圆上的点，则可表示为：$z(x) = e^{i x} = \underbrace{\cos(x)}_{\Re(z)} + \underbrace{i \sin(x)}_{\Im(z)}.$

$x$同时也表示角度$\theta \in \mathbb{R}$，因此式子也可以写为：$z(\theta) = e^{i \theta} = \cos(\theta) + i \sin(\theta).$

### 正弦波

正弦波(sinusoid, 或称正弦曲线)是一种平滑、重复的振荡。
由以下关于时间的连续函数来定义：

$$
y(t) = A \cdot \sin(2\pi ft + \varphi) = A \cdot \sin(\omega t + \varphi)
$$

- $A$<Selector id="sinusoid_A" defaultValue={1.5} options={[0.5, 1, 1.5, 2]} />：振幅。表示函数偏离零点（即其中心值）的最大距离。从图像上看，它决定了波峰和波谷的高度
- $f$：频率。表示每秒发生的完整振荡（或称周期）的次数。其单位通常为赫兹(Hz)
- $\omega=2\pi f$<Selector id="sinusoid_omega" defaultValue={2} options={[-2, -1, 1, 2, 3, 4, 5, 6, 7, 8]} />：角频率。它与频率本质上是一回事，只不过它的通过$\frac{radius}{second}$计算，而不是$\frac{cycles}{second}$
- $\varphi$<Selector id="sinusoid_phi" defaultValue={0} options={[0, "0.5", "1", "1.5"]} />：相位偏移或相移。以弧度为单位。它决定了振荡在$t=0$时是否开始。改变相位，实际上就是将该正弦曲线沿$x$轴（水平轴）向左或向右移动

以$t$为横轴，$y(t)$为纵轴，则可以表示为：$y=y(x)=A \cdot \sin(\omega x+\varphi)$

#### 复正弦波

用振幅$A$乘以欧拉函数，则有：$A*e^{i\theta}=A*(\cos(\theta)+i*\sin(\theta))$

进一步，用随时间变化的相位$\theta=\omega t + \varphi$来替代静态角度$\theta$，就得到了复正弦波(complex sinusoid)：

$$
s(t)=A*e^{i(\omega t + \varphi)} = A * \cos(\underbrace{\omega t + \varphi}_{\theta}) + i * A * \sin(\underbrace{\omega t + \varphi}_{\theta})
$$

复正弦波巧妙地将两个不同的波融合为一个统一的数学实体。随着它随时间演变，它会描绘出一个三维螺旋形状，其中：

- 投影到由$\mathbb{Y}$轴和$\mathbb{Z}$轴张成的平面上，会产生一条正弦波（即虚部）
- 投影到由$\mathbb{X}$轴和$\mathbb{Z}$轴张成的平面上，会产生一条余弦波（即实部）

#### 干涉

当我们把两个或多个正弦波组合在一起时，它们在轴上的每一点的值都会相加。
即：$y = y_1 + y_2 = y_1(x) + y_2(x)$

根据它们的波峰和波谷如何对齐，它们可以相互增强，也可以完全相互抵消——这一现象被称为波的干涉。

两种极端情况：

- 相长干涉(Constructive Interference)：当两个波的波峰与波峰、波谷与波谷对齐时（即同相），振幅相加，产生更强的波。
- 相消干涉(Destructive Interference)：当波峰与波谷对齐时（即反相），振幅相减。如果振幅相等，则完全抵消（合成一条直线）。

    addShowFps(s);

    const w = 800;
    const h = 400;
    const inc = 0.1;
    const sLen = (w - 2 * r) / r / inc;

    const minA = 0.5;
    const maxA = 1;
    const minF = 0.5;
    const maxF = 5;
    const minP = 0;
    const maxP = 2 * 3.14;
    const nSinMax = 7;

    // used to draw the animation
    const states = Object.freeze({
        DROP: 1,
        MERGE: 2,
        STOP: 3
    });

    let mBuff;
    let vC;
    let state = states.DROP;
    let cDrop;
    let cSum;
    let mIdx = 0;
    let nSin = 1;

    // Computes the samples for the dropping sinusoid
    let computeSamples = (dS) => {
        for (let i = 0, cAngl = 0; i < sLen; i += 1, cAngl += inc) {
            dS.samples[i] = s.sin(dS.freq * cAngl + dS.phase) * dS.amp * r;
        }
    }

    let createDrop = (amp, freq, phase) => {
        let r = {};
        r.amp = amp;
        r.freq = freq;
        r.phase = phase;
        r.y = 0;
        r.samples = [];
        computeSamples(r);
        return r;
    }

    // Constructor-like function to create a random sinuscreateDefDrop().samples;oid
    let createRandomDrop = () => {
        return createDrop(
            s.random(minA, maxA),
            s.random(minF, maxF),
            s.random(minP, maxP),
        );
    }

    let createDefDrop = () => {
        return createDrop(1, 1, 0);
    }

    let merge = () => {
        for (let i = 0; i < sLen; i++) {
            cSum[i] += cDrop.samples[i];
        }
    }

    let drawMerge = () => {
        let sum = cDrop.samples[mIdx] + cSum[mIdx];
        let half = h / 2;
        let x1 = 2*r + mIdx * inc * r;
        let y1 = half - cSum[mIdx];
        let x2 = x1;
        let y2 = half - (cSum[mIdx] + (sum - cSum[mIdx]));
        let cColor = (sum > cSum[mIdx]) ? 'green' : 'red';
        mBuff.push();
        mBuff.stroke(cColor);
        mBuff.line(x1, y1, x2, y2);
        mBuff.pop();

        s.stroke('blue');
        s.beginShape();

        s.endShape();
        s.pop();

        s.pop();

        mBuff.push();
        mBuff.noFill();
        mBuff.beginShape();

            let x = 2 * r + cx * r;
            let y = vC.y - cSum[i];

        }
        mBuff.endShape();
        mBuff.pop();

#### 方波

如果精心挑选正弦波并将它们叠加在一起，就能创造出可预测的（几何）图案。

让我们来看一个经典例子：方波。考虑以下无穷级数：

$$
y(x) =  \underbrace{\frac{4}{\pi}\sin(\omega x)}_{y_{1}(x)} + \underbrace{\frac{4}{3\pi}\sin(3\omega x)}_{y_{2}(x)} + \dots + \underbrace{\frac{4}{(2k-1)\pi}{\sin((2k-1)\omega x)}}_{y_k(x)} + \dots
$$

注意$y_1(x), y_2(x), y_3(x), \dots, y_k(x), \dots$都是正弦波，且只加入了奇数次频率($1\omega, 3\omega, \dots$)，而且它们的振幅$A$在按比例递减。

### 本轮

Epicycle(周转圆, 在另一个圆圈上或周围移动的小圆圈)。
古希腊天文学家阿波罗尼提出的用来解释地心体系的一种假想的圆圈。

### 傅里叶级数

#### 谐波分析

上文已可知，任意周期$T$的波$f(t)$都可以以正弦波累加所得。以式子表示即

$$
f(t)=A_0+\sum\limits^\infty_{n=1}A_n\sin(n\omega t+\varphi_n)\;\;\;(n=1,2,3,...)
$$

- 这种展开周期函数的行为被称为：**谐波分析**
- $A_n$、$\varphi_n$，振幅和相移都是已知常数
- $A_0$：直流分量，也是已知常数
- **一次谐波**：$A_1\sin(\omega t+\varphi_1)$
- **二次谐波**：$A_1\sin(2\omega t+\varphi_2)$
- **$n$次谐波**：$A_n\sin(n\omega t+\varphi_n)$

#### 傅里叶系数

由三角公式：$\sin(\alpha\pm\beta)=\sin(\alpha)\cos(\beta)\pm\cos(\alpha)\sin(\beta)$，
变形$n$次谐波：$A_n\sin(n\omega t+\varphi_n)=A_n\sin(\varphi_n)\cos(n\omega t)+A_n\cos(\varphi_n)\sin(n\omega t)$，
由于$A_n$及$\varphi_n$都是常数，因此
记：$a_0=2A_0$(统一分母)，$a_n=A_n\cdot\sin(\varphi_n)$，$b_n=A_n\cdot\cos(\varphi_n)$

则谐波分析的$f(t)$可写成：

$$
f(t)=\frac{a_0}{2}+\sum\limits^\infty_{n=1}\left[a_n\cos(n\omega t)+b_n\sin(n\omega t)\right]
$$

式子中$f(x)$则是**傅里叶级数**，$a_0$、$a_n$、$b_n$为**傅里叶系数**。
[推导](https://zhuanlan.zhihu.com/p/41455378)可得：

- $T$：周期。$T=\frac{1}{f}=\frac{2\pi}{\omega}$
- $a_0=\frac{2}{T}\int_{-\pi}^{\pi}f(t)dt$
- $a_n=\frac{2}{T}\int_{t_0}^{t_0+T}f(t)\cos(\frac{2\pi}{T}tn)dt$
- $b_n=\frac{2}{T}\int_{t_0}^{t_0+T}f(t)\sin(\frac{2\pi}{T}tn)dt$

#### 复指数表示

带入欧拉公式，$f(t)$可[变形](https://zhuanlan.zhihu.com/p/659880737)得：

$$
\begin{gather*}
f(t) = \sum_{n=-N}^{N}C_{n}e^{in\frac{2\pi}{T}t} \\
C_n = \begin{cases}
            \frac{a_0}{2}                   & \text{if } n = 0 \\
            \frac{1}{2} (a_n - i \cdot b_n) & \text{if } n > 0 \\
            \frac{1}{2} (a_n + i \cdot b_n) & \text{if } n < 0 \\
        \end{cases}
\to C_n = \frac{1}{T}\int_{-\frac{T}{2}}^{\frac{T}{2}} f(t)e^{-in\frac{2\pi}{T}t} dt
\end{gather*}
$$

#### 方波的复指数表示

$$
s(t) = \frac{4}{\pi}\sum_{k=1}^{\infty}\frac{\sin((2k-1)\omega t)}{2k-1}
$$

在区间$\left[0, 2L\right]$中，$s(t)$可以写作：

$$
s(t) = 2\left[H(\frac{t}{L})-H(\frac{t}{L}-1)\right] - 1
$$

其中，$H(x)=\begin{cases} 0 & \text{if } x<0\\ 1 & \text{if } x\geq 0\end{cases}$称为阶跃函数(Heavisde Step Function, 赫维赛德阶跃函数)

        };
        for (let i = 1; i < 5; i++) {
            cs[i] = {
                vC: s.createVector(cs[i - 1].vC.x + r / 2, cs[i - 1].vC.y - cs[i - 1].A),
                F: s.PI * (2 * (i + 1) - 1),
                A: 4 / s.PI / (2 * (i + 1) - 1) * r

        buff.textFont(theme.textFont);
        buff.textSize(theme.textSize);

        buff2 = s.createGraphics(w / 2, h);

        });

        buff.line(vC.x + step, vC.y + step, vC.x + 2 * L, vC.y + step);
        buff.pop();

        // Draw sinusoids
        buff.push();
        buff.stroke(theme.sineColor);
        buff.beginShape(s.LINES);
        let prev = s.sqf(0);
        for (let i = 0; i <= 2.01; i += 0.01) {
            let next = s.sqf(i);
            buff.vertex(vC.x + i * L, vC.y - prev);
            buff.vertex(vC.x + i * L, vC.y - next);
            prev = next;
        }
        buff.endShape();
        buff.pop();

        // Draw some labels
        buff.push();
        buff.fill(theme.cosineColor);
        buff.text("f(x)", vC.x + L + 5, vC.y - step - 5);
        buff.fill(theme.sineColor);
        buff.text("∑sᵢ(x)", vC.x + 2 * L + 5, vC.y - 5);
        buff.pop();

        // draw the rectangle surrounding si(x)
        buff2.push();
        buff2.fill(theme.lightGreen);
        buff2.rect(r - r / 2, vG.y - 1.5 * step, 2 * step + r + 10, 3 * L + 30);
        buff2.pop();

        // draw the text
        buff2.push();
        buff2.text(" The first 5 terms of s(x) in time", r - r / 2, vG.y + 1.5 * step + 20);
        buff2.pop();

        // labeling each arrow
        buff2.push();
        for (let i = 0; i < cs.length; i++) {
            buff2.text("s" + (i + 1), cs[i].vC.x - 10, cs[0].vC.y - step * 1.3);
        }
        buff2.pop();

        buff2.push();
        buff2.fill(theme.sineColor);
        buff2.stroke(theme.sineColor);
        buff2.circle(r - r / 2, vG.y + 1.5 * step + 50, 5);
        buff2.fill(theme.cosineColor);
        buff2.stroke(theme.cosineColor);
        buff2.circle(r - r / 2, vG.y + 1.5 * step + 70, 5);
        buff2.pop();

        buff2.push();
        buff2.text(" The sum of the first 5 terms of s(x) in time", r - r / 2 + 10, vG.y + 1.5 * step + 55);
        buff2.pop();

        buff2.push();
        buff2.text(" The square function", r - r / 2 + 10, vG.y + 1.5 * step + 75);
        buff2.pop();
    }

    s.sqf = (x) => {
        let sum = 0;
        for (let i = 0; i < cs.length; i++) {
            sum += cs[i].A * s.sin(cs[i].F * x);
        }
        return sum;
    }

    s.drawCircles = () => {
        s.push();
        s.noFill();
        // update centers
        let ly = 0;
        for (let i = 0; i < cs.length; i++) {
            if (i > 0) {
                cs[i].vC.y = cs[i - 1].vC.y - cs[i - 1].A * s.sin(cs[i - 1].F * angl);
                s.push();
                s.stroke(theme.radiusColorLight);
                s.line(cs[i].vC.x, cs[i].vC.y, cs[i - 1].vC.x, cs[i].vC.y);
                s.pop();
            }
            s.fill(theme.textColor);
            s.circle(cs[i].vC.x, cs[i].vC.y, 3);
            let df = cs[i].A * s.sin(cs[i].F * angl)
            ly += df;
            s.arrowCt(cs[i].vC.x, cs[i].vC.y, cs[i].vC.x, cs[i].vC.y - df, 5);
        }
        s.push();
        s.fill(theme.sineColor);
        s.stroke(theme.sineColor);
        s.circle(vG.x - r, vG.y - ly, 5);
        s.circle(vG.x + angl * step, vG.y - ly, 5);
        s.fill(theme.cosineColor);
        s.stroke(theme.cosineColor);
        let yOff = (angl * step) > L ? -step : step;
        s.circle(vG.x + angl * step, vG.y - yOff, 5);
        s.circle(vG.x - 4 / 5 * step, vG.y - yOff, 5);
        s.pop();
        s.pop();
    }

    s.draw = () => {
        s.background(theme.bkgColor);
        s.image(buff, w / 2, 0);
        s.image(buff2, 0, 0);
        s.drawCircles();
        angl += 0.01;
        if (angl > 2) angl = 0;
        s.showFps();
    }
}

#### 三角波的复指数表示

$$
s(t)=\frac{8}{\pi^2}\sum_{k=1}^{N}\frac{(-1)^{k-1}}{(2k-1)^2}\cdot\sin((2k-1)t)
$$

#### 锯齿波的复指数表示

$$
s(t)=\frac{2}{\pi}\sum_{k=1}^{N}(-1)^k\cdot\frac{\sin(kt)}{k}
$$

const fMachinery = (s) => {
    addPaintGrid(s);
    addArrow(s);

    // the radius of the circle
    const r = 60;
    // the sampling frequency
    const f = 0.1;
    // frequency scaled with r
    const fR = f * r;
    // lol
    const infty = 2000;

            impl: {
                amp: (k) => { return 2 / (s.PI * k) },

                phase: (k) => { return ((k % 2 == 0) ? (0) : (s.PI)); },
            },
            vals: [],
            coeffs: [],
            sum: {},
            maxFreq: 0
